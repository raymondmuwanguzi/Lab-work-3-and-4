using namespace std;

void saveToFile(worker* db, int size) {
ofstream out("workers.dat", ios::binary);
out.write((char*)&size, sizeof(size));

for (int i = 0; i < size; i++)
db[i].save(out);

out.close();
}

worker* loadFromFile(int& size) {
ifstream in("workers.dat", ios::binary);

if (!in) {
size = 0;
return nullptr;
}

in.read((char*)&size, sizeof(size));

worker* db = new worker[size];
for (int i = 0; i < size; i++)
db[i].load(in);

in.close();
return db;
}
